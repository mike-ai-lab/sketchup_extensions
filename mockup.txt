const Slide = ({ tool, index, progress }) => {
  // RECALIBRATED SNAP LOGIC
  // 4 modules = 0.25 progress per module (0, 0.25, 0.5, 0.75, 1.0)
  const snapPoint = index * 0.25;
  
  // Tighter snap detection window for precise triggering
  const snapThreshold = 0.03; // Reduced from 0.08-0.15 for tighter sync
  
  // Global horizontal crawl (all slides move together)
  const x = useTransform(progress, [0, 1], ["0%", "-300%"]);
  
  // PERFECT VERTICAL ALIGNMENT FIX
  // Recalibrated to eliminate drift - content stays centered at snap point
  const contentAlign = useTransform(
    progress,
    [
      snapPoint - 0.125,  // Midpoint between previous snap
      snapPoint,          // Exact snap point
      snapPoint + 0.125   // Midpoint to next snap
    ],
    [30, 0, -30]          // Reduced range for smoother alignment
  );

  // TIGHTENED LIGHT-ON EFFECT
  // Text illuminates ONLY when perfectly snapped
  const textOpacity = useTransform(
    progress,
    [
      snapPoint - snapThreshold * 2,  // Far approach
      snapPoint - snapThreshold,      // Near approach
      snapPoint,                      // SNAP! Full brightness
      snapPoint + snapThreshold,      // Near departure
      snapPoint + snapThreshold * 2   // Far departure
    ],
    [0.2, 0.5, 1, 0.5, 0.2]          // Sharper falloff
  );

  // SYNCHRONIZED SCALE - Peaks exactly at snap
  const boxScale = useTransform(
    progress,
    [
      snapPoint - 0.1,
      snapPoint,
      snapPoint + 0.1
    ],
    [0.88, 1, 0.88]
  );

  // SYNCHRONIZED ROTATION - Neutral exactly at snap
  const boxRotate = useTransform(
    progress,
    [
      snapPoint - 0.2,
      snapPoint,
      snapPoint + 0.2
    ],
    [-12, 0, 12]
  );

  return (
    <motion.div 
      style={{ x }}
      className="relative w-screen h-screen flex-shrink-0 flex items-center justify-center px-10 md:px-24"
    >
      <motion.div 
        style={{ x: contentAlign }}
        className="grid grid-cols-1 lg:grid-cols-12 gap-10 w-full max-w-7xl items-center"
      >
        <div className="lg:col-span-7 space-y-8">
          <div>
             <span className="text-[10px] font-black tracking-[0.6em] text-white/20 uppercase">Module // 0{index + 1}</span>
             <h2 className="text-6xl md:text-9xl font-black italic uppercase tracking-tighter leading-[0.82] mt-4">
               {tool.name.split(' ')[0]}<br/>
               <span style={{ color: tool.accent }}>{tool.name.split(' ')[1]}</span>
             </h2>
          </div>

          <motion.p 
            style={{ opacity: textOpacity }}
            className="text-white max-w-md text-lg italic font-medium transition-opacity duration-100"
          >
            {tool.description}
          </motion.p>

          {/* DEBUG: Show opacity value */}
          <motion.div 
            className="text-[8px] font-mono text-white/30 mt-2"
            style={{ opacity: textOpacity }}
          >
            Opacity: {useTransform(textOpacity, v => v.toFixed(2))}
          </motion.div>

          <div className="flex flex-wrap gap-4">
            {tool.details.map((d, i) => (
              <div key={i} className="px-5 py-2 bg-white/5 border border-white/10 rounded-full text-[9px] font-black uppercase tracking-widest italic text-white/60">
                {d}
              </div>
            ))}
          </div>

          <div className="pt-6">
            <Button className="bg-white text-black hover:bg-white/90 px-12 py-8 rounded-full font-black uppercase tracking-widest text-[10px] group shadow-2xl">
              Load Interface
              <ChevronRight className="ml-2 h-4 w-4 transition-transform group-hover:translate-x-1" />
            </Button>
          </div>
        </div>

        <div className="lg:col-span-5 relative flex justify-center items-center">
           <motion.div 
             style={{ rotate: boxRotate, scale: boxScale }}
             className="relative w-72 h-72 md:w-[450px] md:h-[450px]"
           >
             <div className="absolute inset-0 bg-gradient-to-br from-white/10 to-transparent blur-3xl opacity-20"></div>
             <div className="relative h-full w-full bg-[#0c0c0e] border border-white/5 rounded-[60px] flex items-center justify-center shadow-2xl overflow-hidden">
                <div className="absolute inset-0 opacity-10">
                  <svg width="100%" height="100%"><pattern id={`g-${index}`} width="40" height="40" patternUnits="userSpaceOnUse"><path d="M 40 0 L 0 0 0 40" fill="none" stroke="white" strokeWidth="0.5"/></pattern><rect width="100%" height="100%" fill={`url(#g-${index})`} /></svg>
                </div>
                <Box size={160} strokeWidth={1} style={{ color: tool.accent }} />
             </div>
           </motion.div>
        </div>
      </motion.div>
    </motion.div>
  );
};

  return (

            {[0, 0.25, 0.5, 0.75].map((snap, i) => {
              const isActive = useTransform(
                smoothProgress,
                v => Math.abs(v - snap) < 0.03

              return (
                <motion.div
                  key={i}
                  className="w-3 h-3 rounded-full border-2 border-white/30"

                    backgroundColor: useTransform(
                      smoothProgress,
                      v => Math.abs(v - snap) < 0.03 ? '#3b82f6' : 'transparent'
                    )
                  }}
                />
              );
            })}
          </div>
        </div>
      </motion.div>

      <div ref={scrollContainerRef} className="snap-container relative">

          <div className="sticky top-0 h-screen w-full overflow-hidden flex pointer-events-none">
            {EXTENSIONS.map((tool, index) => (
              <Slide key={tool.id} tool={tool} index={index} progress={smoothProgress} />
            ))}
          </div>
          
          {/* Physical Snap Sections */}
          <div className="absolute top-0 left-0 w-full h-full pointer-events-auto">

              <div key={`snap-${tool.id}`} className="snap-section" />
            ))}
          </div>
        </div>
      </div>

      {/* Footer Meta */}
      <div className="fixed bottom-12 left-12 right-12 z-50 flex justify-between items-end pointer-events-none">
